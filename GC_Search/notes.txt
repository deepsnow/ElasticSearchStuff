how to delete in Sense? "DELETE gc-index"

I can build an index like this in Sense:

PUT gc-index
{
  "settings" : {
    "analysis" : {
      "analyzer" : {
        "remove_html" : {
          "tokenizer" : "standard",
          "char_filter" : [ "html_strip" ]
          }
        }
      }
    }
  
,
  
    "mappings" : {
      "talk" : {
        "properties" : {
          "title" : { "type": "string" },
          "author" : { "type": "string" },
          "confid" : { "type": "string" },
          "content" : { "type": "string", "analyzer": "remove_html" }
        }
      }
    }
  
}
